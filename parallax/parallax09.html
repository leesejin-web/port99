$(window).scroll(function(){
    $(".reveal").each(function(){
        if( $(window).scrollTop() + $(window).height()/2 > $(this).offset().top){
            $(this).delay( $(this).data("delay") ).queue(function(){
                $(this).addClass("show");
            });
        }
    })
})
